function handleSignUp(e) { e.preventDefault(); const user = document.getElementById("signup-username").value.trim(); const email = document.getElementById("signup-email").value.trim(); const pass = document.getElementById("signup-password").value; const confirmPass = document.getElementById("signup-confirm").value; const discordHandle = document.getElementById("signup-discord").value.trim(); if (pass !== confirmPass) { alert("Passwords do not match!"); return; } // Write all user profiles into structured dynamic keys localStorage.setItem("username_" + user, user); localStorage.setItem("password_" + user, pass); localStorage.setItem("email_" + user, email); localStorage.setItem("discord_" + user, discordHandle || "Not Linked"); alert("Account created successfully! Please log in."); toggleForm(false); document.getElementById("login-username").value = user; } function handleLogIn(e) { e.preventDefault(); const user = document.getElementById("login-username").value.trim(); const pass = document.getElementById("login-password").value; const storedUser = localStorage.getItem("username_" + user); const storedPass = localStorage.getItem("password_" + user); if (storedUser === user && storedPass === pass) { localStorage.setItem("isLoggedIn", "true"); // Store session identity tracker token localStorage.setItem("activeSessionUser", user); window.location.href = "index.html"; } else { alert("Invalid username or password!"); } }
